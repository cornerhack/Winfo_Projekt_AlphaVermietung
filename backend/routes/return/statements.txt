//// Nach log in möglich

SELECT mietvertragID
FROM mietvertraege
WHERE reservierungID IN (
  SELECT reservierungID
  FROM reservierungen
  WHERE kundeID = 1
);


/// Suche nach information mit vertrags id


SELECT
  marke,
  modell,
  kilometerStand 
FROM
  kfz
WHERE kfztypID IN (
    SELECT kfztypID
    FROM reservierungen
    WHERE reservierungID IN (
        SELECT reservierungID
        FROM mietvertraege
        WHERE mietvertragID = 1;
    )
);

/// Suche nach  gemieteten Autos (Autorückname) 

SELECT

k.marke,

k.modell,

k.kilometerStand

FROM kfz k

WHERE k.kfzTypID IN (

SELECT r.kfzTypID

FROM reservierungen r

WHERE r.reservierungID IN (

    SELECT m.reservierungID

    FROM mietvertraege m

        WHERE m.mietvertragID IN (

        SELECT m2.mietvertragID

        FROM mietvertraege m2

        JOIN reservierungen r2 ON m2.reservierungID = r2.reservierungID

        WHERE r2.kundeID = 1

    )

) ) ;
